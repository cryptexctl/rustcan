```rust
pub fn handle_error(error: ScanError) -> Result<()> {
    match error {
        ScanError::ConnectionError(e) => {
            tracing::warn!("Ошибка соединения: {}", e);
            Ok(())
        },
        ScanError::TimeoutError(msg) => {
            tracing::warn!("Таймаут: {}", msg);
            Ok(())
        },
        _ => Err(error.into()),
    }
}
```
